cmake_minimum_required(VERSION "3.21")
project(Surviving_Maps_Ingest VERSION 0.0.0)

set(CPACK_PROJECT_NAME ${PROJECT_NAME}) 
set(CPACK_PROJECT_VERSION ${PROJECT_VERSION}) 
include(CPack)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_INCLUDE_CURRENT_DIR ON)

file(GLOB_RECURSE SOURCES "Source/*.cpp")

add_executable(Surviving_Maps_Ingest ${SOURCES})

target_include_directories(Surviving_Maps_Ingest PRIVATE ${CMAKE_CURRENT_SOURCE_DIR}/Include)
target_include_directories(Surviving_Maps_Ingest PRIVATE ${CMAKE_CURRENT_SOURCE_DIR}/Include/Site)
target_include_directories(Surviving_Maps_Ingest PRIVATE ${CMAKE_CURRENT_SOURCE_DIR}/Include/VariantData)
target_include_directories(Surviving_Maps_Ingest PRIVATE ${CMAKE_CURRENT_SOURCE_DIR}/Include/SearchContext)

add_custom_command(TARGET Surviving_Maps_Ingest POST_BUILD
	COMMAND ${CMAKE_COMMAND} -E copy_directory_if_different
	"${PROJECT_SOURCE_DIR}/CSVMapData/"
	"${CMAKE_CURRENT_BINARY_DIR}/CSVMapData/")
